1676 // An "all-purpose" add/subtract immediate, per ARM documentation:
1677 // A "programmer-friendly" assembler may accept a negative immediate
1678 // between -(2^24 -1) and -1 inclusive, causing it to convert a
1679 // requested ADD operation to a SUB, or vice versa, and then encode
1680 // the absolute value of the immediate as for uimm24.
1681 void Assembler::add_sub_immediate(Register Rd, Register Rn, unsigned uimm, int op,
1682                                   int negated_op) {
1683   bool sets_flags = op & 1;   // this op sets flags
1684   union {
1685     unsigned u;
1686     int imm;
1687   };
1688   u = uimm;
1689   bool shift = false;
1690   bool neg = imm < 0;
1691   if (neg) {
1692     imm = -imm;
1693     op = negated_op;
1694   }
1695   assert(Rd != sp || imm % 16 == 0, "misaligned stack");
1696   if (imm >= (1 << 11)
1697       && ((imm >> 12) << 12 == imm)) {
1698     imm >>= 12;
1699     shift = true;
1700   }
1701   f(op, 31, 29), f(0b10001, 28, 24), f(shift, 23, 22), f(imm, 21, 10);
1702 
1703   // add/subtract immediate ops with the S bit set treat r31 as zr;
1704   // with S unset they use sp.
1705   if (sets_flags)
1706     zrf(Rd, 0);
1707   else
1708     srf(Rd, 0);
1709 
1710   srf(Rn, 5);
1711 }
