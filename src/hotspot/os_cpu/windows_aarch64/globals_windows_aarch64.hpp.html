29 // Sets the default values for platform dependent flags used by the runtime system.
30 // (see globals.hpp)
31 
32 define_pd_global(bool, DontYieldALot,            false);
33 
34 // Default stack size on Windows is determined by the executable (java.exe
35 // has a default value of 320K/1MB [32bit/64bit]). Depending on Windows version, changing
36 // ThreadStackSize to non-zero may have significant impact on memory usage.
37 // See comments in os_windows.cpp.
38 define_pd_global(intx, ThreadStackSize,          0); // 0 => use system default
39 define_pd_global(intx, VMThreadStackSize,        0);
40 
41 #ifdef ASSERT
42 define_pd_global(intx, CompilerThreadStackSize,  1024);
43 #else
44 define_pd_global(intx, CompilerThreadStackSize,  0);
45 #endif
46 
47 define_pd_global(uintx,JVMInvokeMethodSlack,     8192);
48 
49 // Used on 64 bit platforms for UseCompressedOops base address
50 define_pd_global(uintx,HeapBaseMinAddress,       2*G);
