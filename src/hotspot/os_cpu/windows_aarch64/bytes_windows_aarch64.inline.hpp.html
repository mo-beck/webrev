26 #ifndef OS_CPU_WINDOWS_AARCH64_BYTES_WINDOWS_AARCH64_INLINE_HPP
27 #define OS_CPU_WINDOWS_AARCH64_BYTES_WINDOWS_AARCH64_INLINE_HPP
28 
29 #include <stdlib.h>
30 
31 // Efficient swapping of data bytes from Java byte
32 // ordering to native byte ordering and vice versa.
33 inline u2   Bytes::swap_u2(u2 x) {
34   return _byteswap_ushort(x);
35 }
36 
37 inline u4   Bytes::swap_u4(u4 x) {
38   return _byteswap_ulong(x);
39 }
40 
41 inline u8 Bytes::swap_u8(u8 x) {
42   return _byteswap_uint64(x);
43 }
44 
45 #pragma warning(default: 4035) // Enable warning 4035: no return value
46 
47 #endif // OS_CPU_WINDOWS_AARCH64_BYTES_WINDOWS_AARCH64_INLINE_HPP
