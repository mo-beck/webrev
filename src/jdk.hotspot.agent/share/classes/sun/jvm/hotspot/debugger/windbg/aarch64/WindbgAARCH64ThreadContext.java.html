25 package sun.jvm.hotspot.debugger.windbg.aarch64;
26 
27 import sun.jvm.hotspot.debugger.*;
28 import sun.jvm.hotspot.debugger.aarch64.*;
29 import sun.jvm.hotspot.debugger.windbg.*;
30 
31 class WindbgAARCH64ThreadContext extends AARCH64ThreadContext {
32   private WindbgDebugger debugger;
33 
34   public WindbgAARCH64ThreadContext(WindbgDebugger debugger) {
35     super();
36     this.debugger = debugger;
37   }
38 
39   public void setRegisterAsAddress(int index, Address value) {
40     setRegister(index, debugger.getAddressValue(value));
41   }
42 
43   public Address getRegisterAsAddress(int index) {
44     return debugger.newAddress(getRegister(index));
45   }
46 }
