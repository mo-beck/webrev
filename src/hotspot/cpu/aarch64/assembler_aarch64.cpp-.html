1670 // An "all-purpose" add/subtract immediate, per ARM documentation:
1671 // A "programmer-friendly" assembler may accept a negative immediate
1672 // between -(2^24 -1) and -1 inclusive, causing it to convert a
1673 // requested ADD operation to a SUB, or vice versa, and then encode
1674 // the absolute value of the immediate as for uimm24.
1675 void Assembler::add_sub_immediate(Register Rd, Register Rn, unsigned uimm, int op,
1676                                   int negated_op) {
1677   bool sets_flags = op & 1;   // this op sets flags
1678   union {
1679     unsigned u;
1680     int imm;
1681   };
1682   u = uimm;
1683   bool shift = false;
1684   bool neg = imm < 0;
1685   if (neg) {
1686     imm = -imm;
1687     op = negated_op;
1688   }
1689   assert(Rd != sp || imm % 16 == 0, "misaligned stack");
1690   if (imm >= (1 << 11)
1691       && ((imm >> 12) << 12 == imm)) {
1692     imm >>= 12;
1693     shift = true;
1694   }
1695   f(op, 31, 29), f(0b10001, 28, 24), f(shift, 23, 22), f(imm, 21, 10);
1696 
1697   // add/subtract immediate ops with the S bit set treat r31 as zr;
1698   // with S unset they use sp.
1699   if (sets_flags)
1700     zrf(Rd, 0);
1701   else
1702     srf(Rd, 0);
1703 
1704   srf(Rn, 5);
1705 }
