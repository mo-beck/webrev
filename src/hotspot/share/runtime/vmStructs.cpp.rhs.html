3129   {
3130     VMTypeEntry* types = origtypes;

3133         // Found it

3136       ++types;
3137     }
3138   }
3139   // Search for the base type by peeling off const and *
3140   size_t len = strlen(typeName);
3141   if (typeName[len-1] == '*') {
3142     char * s = NEW_C_HEAP_ARRAY(char, len, mtInternal);
3143     strncpy(s, typeName, len - 1);
3144     s[len-1] = '\0';

3146     if (recursiveFindType(origtypes, s, true) == 1) {
3147       FREE_C_HEAP_ARRAY(char, s);
3148       return 1;
